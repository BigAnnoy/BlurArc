```python
# backend/api_server.py

@app.route('/api/photos/<int:photo_id>/favorite', methods=['POST'])
def add_favorite(photo_id):
    """收藏照片"""
    db = SessionLocal()
    try:
        photo = db.query(Photo).filter(Photo.id == photo_id).first()
        if not photo:
            return jsonify({'error': 'Photo not found'}), 404
        
        photo.is_favorite = True
        photo.favorited_at = datetime.now()
        db.commit()
        
        return jsonify({'success': True, 'favorited_at': photo.favorited_at.isoformat()})
    finally:
        db.close()

@app.route('/api/photos/<int:photo_id>/favorite', methods=['DELETE'])
def remove_favorite(photo_id):
    """取消收藏"""
    db = SessionLocal()
    try:
        photo = db.query(Photo).filter(Photo.id == photo_id).first()
        if not photo:
            return jsonify({'error': 'Photo not found'}), 404
        
        photo.is_favorite = False
        photo.favorited_at = None
        db.commit()
        
        return jsonify({'success': True})
    finally:
        db.close()

@app.route('/api/photos/batch-favorite', methods=['POST'])
def batch_favorite():
    """批量收藏/取消收藏"""
    data = request.json
    photo_ids = data.get('photo_ids', [])
    favorite = data.get('favorite', True)
    
    db = SessionLocal()
    try:
        photos = db.query(Photo).filter(Photo.id.in_(photo_ids)).all()
        now = datetime.now() if favorite else None
        
        for photo in photos:
            photo.is_favorite = favorite
            photo.favorited_at = now
        
        db.commit()
        return jsonify({'success': True, 'updated': len(photos)})
    finally:
        db.close()

@app.route('/api/photos/favorites', methods=['GET'])
def get_favorites():
    """获取收藏列表"""
    db = SessionLocal()
    try:
        photos = db.query(Photo).filter(Photo.is_favorite == True).order_by(Photo.favorited_at.desc()).all()
        
        result = []
        for p in photos:
            result.append({
                'id': p.id,
                'filename': p.filename,
                'path': p.path,
                'size': p.size,
                'date': p.media_date.isoformat() if p.media_date else None,
                'type': p.file_type,
                'is_favorite': p.is_favorite,
                'favorited_at': p.favorited_at.isoformat() if p.favorited_at else None,
            })
        
        return jsonify({'photos': result, 'total': len(result)})
    finally:
        db.close()
```

```tsx
// frontend/src/components/dialogs/AlbumManageModal.tsx

import { useState } from 'react';
import { Modal } from '../common/Modal';
import { api, Album } from '../../services/api';

interface AlbumManageModalProps {
  isOpen: boolean;
  onClose: () => void;
  album?: Album;
  onSaved: () => void;
}

export function AlbumManageModal({ isOpen, onClose, album, onSaved }: AlbumManageModalProps) {
  const [name, setName] = useState(album?.name || '');
  
  const handleSave = async () => {
    if (!name.trim()) return;
    
    if (album) {
      await api.updateAlbum(album.id, name);
    } else {
      await api.createAlbum(name);
    }
    
    onSaved();
    onClose();
  };
  
  return (
    <Modal isOpen={isOpen} onClose={onClose} title={album ? '重命名相册' : '新建相册'}>
      <div className="p-6">
        <input
          type="text"
          value={name}
          onChange={(e) => setName(e.target.value)}
          placeholder="相册名称"
          className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
          autoFocus
```

```python
# backend/xmp_sync.py

from pathlib import Path
from libxmp import XMPFiles, XMPMeta

def sync_description_to_xmp(photo_path: str, description: str):
    """将描述同步到 XMP"""
    xmp_file = XMPFiles(file_path=photo_path, open_forupdate=True)
    xmp = xmp_file.get_xmp()
    
    if xmp:
        xmp.set_property(XMPConst.NS_DC, 'description', description)
        xmp_file.put_xmp(xmp)
    
    xmp_file.close_file()

def sync_description_from_xmp(photo_path: str) -> str:
    """从 XMP 读取描述"""
    xmp_file = XMPFiles(file_path=photo_path)
    xmp = xmp_file.get_xmp()
    
    description = ''
    if xmp:
        description = xmp.get_property(XMPConst.NS_DC, 'description') or ''
    
    xmp_file.close_file()
    return description
```

```python
# backend/api_server.py

@app.route('/api/photos/<int:photo_id>/description', methods=['PUT'])
def update_description(photo_id):
    """更新照片描述（同步到 XMP）"""
    data = request.json
    description = data.get('description', '')
    
    db = SessionLocal()
    try:
        photo = db.query(Photo).filter(Photo.id == photo_id).first()
        if not photo:
            return jsonify({'error': 'Photo not found'}), 404
        
        # 更新 DB
        photo.description = description
        db.commit()
        
        # 同步到 XMP
        sync_description_to_xmp(photo.path, description)
        
        return jsonify({'success': True})
    finally:
        db.close()
```
