class TestThumbnailCache:
    """测试缩略图缓存命中/失效逻辑"""

    def test_first_generation_writes_to_cache(self, tmp_path, make_image, temp_thumb_dir):
        from backend.thumbnail_manager import ThumbnailManager
        mgr = ThumbnailManager()
        img = make_image(tmp_path / "src.jpg", 200, 200)
        # 强制使用 temp 目录
        mgr.cache_dir = temp_thumb_dir
        result = mgr.generate_image_thumb(img, size=128)
        assert result is not None
        assert len(result) > 0
        # 缓存文件应被创建
        cache_files = list(temp_thumb_dir.iterdir())
        assert len(cache_files) >= 1

    def test_second_access_hits_cache(self, tmp_path, make_image, temp_thumb_dir, monkeypatch):
        from backend.thumbnail_manager import ThumbnailManager
        mgr = ThumbnailManager()
        mgr.cache_dir = temp_thumb_dir
        img = make_image(tmp_path / "src.jpg", 200, 200)

        # 第一次生成
        result1 = mgr.generate_image_thumb(img, size=128)

        # 监控第二次的 stat 调用
        stat_called = []
        original_stat = Path.stat
        def counting_stat(self, *args, **kwargs):
            stat_called.append(self)
            return original_stat(self, *args, **kwargs)
        monkeypatch.setattr(Path, "stat", counting_stat)

        # 第二次（应命中缓存）
        result2 = mgr.generate_image_thumb(img, size=128)
        assert result1 == result2
        # 缓存命中时不应再 stat 源文件
        assert len(stat_called) == 0, f"Expected 0 stat calls, got {len(stat_called)}"

    def test_cache_invalidated_on_mtime_change(self, tmp_path, make_image, temp_thumb_dir):
        from backend.thumbnail_manager import ThumbnailManager
        mgr = ThumbnailManager()
        mgr.cache_dir = temp_thumb_dir
        img = make_image(tmp_path / "src.jpg", 200, 200)
        mgr.generate_image_thumb(img, size=128)
        # 改 mtime
        os.utime(img, (1000000, 1000000))
        # 删除缓存条目（模拟 mtime 变化）
        cache_files_before = len(list(temp_thumb_dir.iterdir()))
        mgr.generate_image_thumb(img, size=128)
        # 缓存应被更新
        assert len(list(temp_thumb_dir.iterdir())) >= cache_files_before

    def test_concurrent_same_file_generates_once(self, tmp_path, make_image, temp_thumb_dir):
