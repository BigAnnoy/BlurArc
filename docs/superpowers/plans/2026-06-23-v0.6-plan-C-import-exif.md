```python
# 性能优化（v0.6 优化 4）：EXIF magic bytes 白名单
# 命中才开 PIL 解析，截图/网络图/无 EXIF 文件直接 mtime fallback
EXIF_SUPPORTED_MAGIC = frozenset({
    b'\xff\xd8\xff',       # JPEG
    b'\x89PNG',            # PNG
    b'RIFF',               # WebP（前 4 字节 RIFF）
    b'II*\x00',            # TIFF (little-endian)
    b'MM\x00*',            # TIFF (big-endian)
})


def _has_exif_magic(path: Path) -> bool:
    """快速检测文件头是否为已知图片格式（无 EXIF 时直接走 mtime）"""
    try:
        with open(path, 'rb') as f:
            return f.read(4) in EXIF_SUPPORTED_MAGIC
    except OSError:
        return False
```

```python
def _get_media_date(file_path: Path):
    """提取文件媒体日期（优先 EXIF，回退 mtime）"""
    # 优化 4 前置：magic bytes 不在白名单 → 直接 mtime
    if not _has_exif_magic(file_path):

            return datetime.fromtimestamp(file_path.stat().st_mtime)

            return None

        from PIL import Image

        exif = img._getexif() if hasattr(img, '_getexif') else None
        if exif:

        return datetime.fromtimestamp(file_path.stat().st_mtime)
    except OSError:
        return None
```

```python
"""_has_exif_magic 专项测试"""
import pytest
from pathlib import Path

from backend.import_manager import _has_exif_magic, EXIF_SUPPORTED_MAGIC


class TestExifMagicBytes:
    """magic bytes 白名单测试"""

    @pytest.mark.parametrize("magic,expected", [
        (b'\xff\xd8\xff\xe0', True),   # JPEG/JFIF
        (b'\xff\xd8\xff\xe1', True),   # JPEG/Exif
        (b'\x89PNG\r\n', True),        # PNG
        (b'RIFF\x00\x00\x00\x00WEBP', True),  # WebP (RIFF container)
        (b'II*\x00', True),            # TIFF little-endian
        (b'MM\x00*', True),            # TIFF big-endian
        (b'NOT_MAGIC', False),
        (b'PK\x03\x04', False),        # ZIP
        (b'%PDF', False),              # PDF
        (b'', False),                  # empty
    ])
    def test_magic_detection(self, tmp_path, magic, expected):
        f = tmp_path / "test"
        f.write_bytes(magic + b"\x00" * 100)
        assert _has_exif_magic(f) == expected

    def test_jpeg_with_exif_date_found(self, tmp_path):
        """有 EXIF 的 JPEG 应通过 magic check"""
        from PIL import Image
        from piexif import create_exif, ExifIFD
        img_path = tmp_path / "with_exif.jpg"
        img = Image.new("RGB", (100, 100))
        exif = create_exif({"0th": {}, "Exif": {}, "GPS": {}, "1st": {},
                           "thumbnail": None})
        exif["Exif"][ExifIFD.DateTimeOriginal] = b"2023:05:15 14:30:00"
        img.save(img_path, exif=exif)
        assert _has_exif_magic(img_path) is True

    def test_png_passes_magic(self, tmp_path):
        """PNG 应通过 magic check（即使无 EXIF）"""
        from PIL import Image
        img_path = tmp_path / "test.png"
        Image.new("RGB", (100, 100)).save(img_path)
        assert _has_exif_magic(img_path) is True

    def test_text_file_rejected(self, tmp_path):
        """文本文件应被拒绝"""
        f = tmp_path / "test.txt"
        f.write_text("not an image")
        assert _has_exif_magic(f) is False

        """不存在文件不应抛错"""
```
