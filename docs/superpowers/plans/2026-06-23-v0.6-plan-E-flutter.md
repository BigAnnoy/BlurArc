import 'package:flutter_test/flutter_test.dart';
import 'package:mocktail/mocktail.dart';
import 'package:dio/dio.dart';

import 'package:blurarc_app/services/api_client.dart';
import '../fixtures/mock_responses.dart';

class MockDio extends Mock implements Dio {}

void main() {
  late MockDio mockDio;
  late ApiClient apiClient;

  setUp(() {
    mockDio = MockDio();
    apiClient = ApiClient(dio: mockDio);
  });

  group('ApiClient.fetchPhotos', () {
    test('成功返回照片列表', () async {
      // Arrange
      when(() => mockDio.get(any())).thenAnswer(
        (_) async => Response(
          data: {'photos': [], 'total': 0},
          statusCode: 200,
          requestOptions: RequestOptions(path: '/api/photos'),
        ),
      );

      // Act
      final result = await apiClient.fetchPhotos();

      // Assert
      expect(result, isA<List<Photo>>());
      verify(() => mockDio.get('/api/photos')).called(1);
    });

    test('空列表处理', () async {
      when(() => mockDio.get(any())).thenAnswer(
        (_) async => Response(
          data: {'photos': null, 'total': 0},
          statusCode: 200,
          requestOptions: RequestOptions(path: '/api/photos'),
        ),
      );

      final result = await apiClient.fetchPhotos();
      expect(result, isEmpty);
    });

    test('网络错误抛出 ApiException', () async {
      when(() => mockDio.get(any())).thenThrow(
        DioException(
          requestOptions: RequestOptions(path: '/api/photos'),
          type: DioExceptionType.connectionTimeout,
        ),
      );

      expect(apiClient.fetchPhotos(), throwsA(isA<ApiException>()));
    });

    test('HTTP 500 抛出错误', () async {
      when(() => mockDio.get(any())).thenAnswer(
        (_) async => Response(
          data: {'error': 'Internal Server Error'},
          statusCode: 500,
          requestOptions: RequestOptions(path: '/api/photos'),
        ),
      );

      expect(apiClient.fetchPhotos(), throwsA(isA<ApiException>()));
    });
  });

  group('ApiClient.fetchPhoto', () {
    test('成功返回单张照片详情', () async {
      when(() => mockDio.get('/api/photos/1')).thenAnswer(
        (_) async => Response(

            'filename': 'test.jpg',
            'path': '/test.jpg',
            'size': 1000,
          },
          statusCode: 200,
          requestOptions: RequestOptions(path: '/api/photos/1'),
        ),
      );

      final photo = await apiClient.fetchPhoto(1);
      expect(photo.id, 1);
      expect(photo.filename, 'test.jpg');
    });
  });
}

import 'package:flutter_test/flutter_test.dart';
import 'package:mocktail/mocktail.dart';
import 'package:shared_preferences/shared_preferences.dart';

import 'package:blurarc_app/services/pairing_service.dart';
import 'package:blurarc_app/services/api_client.dart';
import '../fixtures/mock_responses.dart';

class MockApiClient extends Mock implements ApiClient {}

void main() {
  late MockApiClient mockApi;
  late PairingService service;

  setUp(() async {
    SharedPreferences.setMockInitialValues({});
    mockApi = MockApiClient();
    service = PairingService(apiClient: mockApi);
  });

  group('PairingService.pair', () {
    test('正确配对码返回 token', () async {
      when(() => mockApi.pair(
            pairingCode: any(named: 'pairingCode'),
            deviceName: any(named: 'deviceName'),
          )).thenAnswer((_) async => PairingResult(
            success: true,
            token: 'test_token_abc123',
            deviceId: 1,
          ));

      final result = await service.pair(
        pairingCode: '1234',
        deviceName: 'iPhone-Test',
      );

      expect(result.success, isTrue);
      expect(result.token, 'test_token_abc123');
    });

    test('错误配对码抛出 PairingException', () async {
      when(() => mockApi.pair(
            pairingCode: any(named: 'pairingCode'),
            deviceName: any(named: 'deviceName'),
          )).thenAnswer((_) async => PairingResult(
            success: false,
            error: 'Invalid pairing code',
          ));

      expect(
        service.pair(pairingCode: '0000', deviceName: 'Test'),
        throwsA(isA<PairingException>()),
      );
    });

    test('配对成功后 token 被持久化', () async {
      when(() => mockApi.pair(
            pairingCode: any(named: 'pairingCode'),
            deviceName: any(named: 'deviceName'),
          )).thenAnswer((_) async => PairingResult(
            success: true,
            token: 'persistent_token',
            deviceId: 1,
          ));

      await service.pair(pairingCode: '1234', deviceName: 'Test');
      final prefs = await SharedPreferences.getInstance();
      expect(prefs.getString('pairing_token'), 'persistent_token');
    });
  });

  group('PairingService.isPaired', () {
    test('无 token 时返回 false', () async {
      expect(await service.isPaired(), isFalse);
    });

    test('有 token 时返回 true', () async {
      SharedPreferences.setMockInitialValues({
        'pairing_token': 'existing_token',
      });
      expect(await service.isPaired(), isTrue);
    });
  });
}

    service = UploadService(apiClient: mockApi);
    testFile = File('test/fixtures/sample.jpg');
    testFile.parent.createSync(recursive: true);
    testFile.writeAsBytesSync(List.filled(1024, 0xFF));
  });

  tearDown(() {
    if (testFile.existsSync()) testFile.deleteSync();
  });

  test('上传单张图片成功', () async {
    when(() => mockApi.uploadFile(
          file: any(named: 'file'),
          onProgress: any(named: 'onProgress'),
        )).thenAnswer((_) async => UploadResult(success: true, filename: 'uploaded.jpg'));

    final result = await service.uploadImage(testFile);
    expect(result.success, isTrue);
    expect(result.filename, 'uploaded.jpg');
  });

  test('上传进度回调被调用', () async {
    final progressValues = <double>[];
    when(() => mockApi.uploadFile(
          file: any(named: 'file'),
          onProgress: any(named: 'onProgress'),
        )).thenAnswer((invocation) async {
      final cb = invocation.namedArguments[#onProgress] as Function?;
      cb?.call(0.5);
      cb?.call(1.0);
      return UploadResult(success: true, filename: 'p.jpg');
    });

    await service.uploadImage(
      testFile,
      onProgress: (p) => progressValues.add(p),
    );
    expect(progressValues, [0.5, 1.0]);
  });

  test('文件不存在抛出 FileSystemException', () async {
    expect(
      service.uploadImage(File('nonexistent.jpg')),
      throwsA(isA<FileSystemException>()),
    );
  });
}
