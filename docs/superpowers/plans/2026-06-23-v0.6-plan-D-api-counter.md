```python
"""
目标文件名生成（含原子计数器）
- 进程级 itertools.count() 计数器
- 多线程场景下配合 Lock 避免 O(n) 探测
"""
import itertools
import threading
from pathlib import Path
from typing import Optional


# 进程级原子计数器（从 1 开始）
_seq_counter = itertools.count(1)

# 用于保护 taken_names 的锁
_filename_lock = threading.Lock()


def reset_counter() -> None:
    """重置计数器（仅测试用）"""
    global _seq_counter
    _seq_counter = itertools.count(1)


def _next_seq() -> int:
    """获取下一个序号（O(1)）"""
    return next(_seq_counter)


def build_dest_filename(
    media_date,
    ext: str,
    month_dir: Path,
    is_dup: bool,
    taken_names: set,
) -> Path:
    """
    生成目标文件名（线程安全，O(1)）

    Args:
        media_date: datetime 对象
        ext: 扩展名（.jpg）
        month_dir: 目标月份目录
        is_dup: 是否为重复文件
        taken_names: 已被占用的文件名集合（线程间共享）

    Returns:
        完整的目标文件路径
    """
    stem = media_date.strftime("%Y%m%d_%H%M%S")
    with _filename_lock:
        while True:
            n = _next_seq()
            if is_dup:
                candidate = f"{stem}_{n:03d}_dup{ext}"
            else:
                candidate = f"{stem}_{n:03d}{ext}"
            if candidate not in taken_names:
                taken_names.add(candidate)
                return month_dir / candidate
```

                datetime(2024, 1, 1, 12, 0, 0),
                ".jpg", tmp_path, False, taken
            )
            with taken_lock:
                results.append(path.name)

        threads = [threading.Thread(target=generate, args=(i,)) for i in range(100)]
        for t in threads:
            t.start()
        for t in threads:
            t.join()

        # 100 个文件名应全部唯一
        assert len(results) == 100
        assert len(set(results)) == 100, f"Collisions detected: {len(results) - len(set(results))}"

    def test_counter_persists_across_calls(self, tmp_path):
        """计数器应保持递增"""
        for _ in range(10):
            build_dest_filename(
                datetime(2024, 1, 1), ".jpg", tmp_path, False, set()
            )
        # 内部计数器已递增到 10，下一次应从 11 开始
        assert _next_seq() == 11

"""完整导入流程 API 集成测试"""
import pytest
import time


class TestFullImportWorkflow:
    """check → start → progress → list 完整链路"""

    def test_check_then_start_then_progress(self, client, sample_album, tmp_path):
        target = tmp_path / "target"

        # 1. check
        resp = client.post("/api/import/check", json={
            "source": str(sample_album),
            "ignore_last_scan": True,
        })
        assert resp.status_code == 200
        assert resp.json.get("total_files", 0) == 10

        # 2. start
        resp = client.post("/api/import/start", json={
            "source": str(sample_album),
            "target": str(target),
            "ignore_last_scan": True,
        })
        assert resp.status_code in [200, 202]
        task_id = resp.json.get("import_id") or resp.json.get("task_id")

        # 3. progress
        if task_id:
            for _ in range(30):
                resp = client.get("/api/import/progress",
                                query_string={"task_id": task_id})
                if resp.json.get("status") in ["completed", "done", "finished"]:
                    break
                time.sleep(0.5)

        # 4. 验证目标目录有文件
        target_files = list(target.rglob("*.jpg"))
        assert len(target_files) == 10, f"Expected 10 files, got {len(target_files)}"
        assert len(set(target_files)) == 10  # 无重复

    def test_naming_pattern_yyyymm(self, client, sample_album, tmp_path):
        """验证导入后路径是 YYYY/YYYY-MM/"""
        target = tmp_path / "target"
        resp = client.post("/api/import/start", json={
            "source": str(sample_album),
            "target": str(target),
            "ignore_last_scan": True,
        })
        assert resp.status_code in [200, 202]
        # 等待完成
        time.sleep(2)

        # 应有 YYYY/YYYY-MM/ 子目录
        year_dirs = [d for d in target.iterdir() if d.is_dir()]
        assert len(year_dirs) >= 1
        for yd in year_dirs:
            month_dirs = [d for d in yd.iterdir() if d.is_dir()]
            assert len(month_dirs) >= 1
            # 路径格式应是 YYYY-MM
            assert "-" in month_dirs[0].name
